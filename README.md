# tmai-spring
Spring Boot API

mvn clean install

# git commands
# to fetch git history
git fetch --all

# To checkou to any branch of already created in github
git checkout -m feature/first-branch (example)
 
# To check if any file changes
git status

# adding all files or changes
git add .

# To add each file one by one
git add read.md

# Commiting the changes to branch
git commit -m "Commit messages"

# Pushing changes to feature/develop branch
git push 

or 

git push origin develop
